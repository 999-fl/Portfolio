Important code (inverted index snippet)

// build inverted index from notes
const buildIndex = (notes) => {
  const idx = {};
  notes.forEach((note, id) => {
    const words = note.text.toLowerCase().match(/\w+/g) || [];
    new Set(words).forEach(w => (idx[w] = idx[w] || new Set()).add(id));
  });
  // convert sets to arrays for JSON
  return Object.fromEntries(Object.entries(idx).map(([k,s]) => [k, [...s]]));
};
